#include
#include
using namespace std;
int	n, m,num;
vector  list;
int c(int a) {
	list.push_back(a);
	num--;
	if (num >= 0) {
		for (int i = a + 1; i <= n; i++) {
			c(i);
		}
	}
	if (num ==0) {
		for (int i = 0; i < list.size(); i++) {
			cout << list[i] <<" ";
		}
		cout << endl;
	}
	list.pop_back();
	num++;
	return 0;
}
int main(){
	while (cin>>n>>m) {
		if (m > n)continue;
		for (int i = 1; i <= n-m+1; i++) {
			list.clear();
			num = m;
			c(i);
		}
	}
}
